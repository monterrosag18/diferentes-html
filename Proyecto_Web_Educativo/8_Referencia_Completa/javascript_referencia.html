Caso Pr√°ctico: Carrito Persistente

Ver c√≥digo JavaScript 
// Estructura de carrito
let carrito = {
    items: [],
    total: 0
};

// Funci√≥n para agregar al carrito
function agregarAlCarrito(producto, precio) {
    // Cargar carrito guardado
    let carritoGuardado = JSON.parse(localStorage.getItem("carrito")) || carrito;
    
    // Agregar nuevo item
    carritoGuardado.items.push({
        producto: producto,
        precio: precio
    });
    
    // Calcular total
    carritoGuardado.total = carritoGuardado.items.reduce((sum, item) => {
        return sum + item.precio;
    }, 0);
    
    // Guardar
    localStorage.setItem("carrito", JSON.stringify(carritoGuardado));
    console.log("Producto agregado");
}

// Funci√≥n para obtener carrito
function obtenerCarrito() {
    let carritoGuardado = localStorage.getItem("carrito");
    if (!carritoGuardado) {
        return carrito;
    }
    return JSON.parse(carritoGuardado);
}

// Funci√≥n para vaciar carrito
function vaciarCarrito() {
    localStorage.removeItem("carrito");
    console.log("Carrito vaciado");
}

// Usar las funciones
agregarAlCarrito("Laptop", 1000);
agregarAlCarrito("Mouse", 20);

let mi_carrito = obtenerCarrito();
console.log(mi_carrito);  // Ver carrito completo
console.log(mi_carrito.total);  // Ver total
